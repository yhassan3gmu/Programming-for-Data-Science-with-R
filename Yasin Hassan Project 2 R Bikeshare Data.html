#this calculates the mode for each month
mode_month <- function(months){
    uniqm<- unique(months)
    uniqm[which.max(tabulate(match(months,uniqm)))]
}

cities<- list(ny, wash, chi)
for (CITY in cities){
    #extract month from Start.Time
    CITY$month = format(as.Date(CITY$Start.Time, format = '%Y-%m-%d %H:%M:%S'), '%b')
    #print data for each month
    print(aggregate(CITY$month, by=list(CITY$month), FUN=length))
    #print the most common month
    print(mode_month(CITY$month))
}
